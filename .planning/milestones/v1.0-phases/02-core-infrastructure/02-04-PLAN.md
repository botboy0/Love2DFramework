end

**Fully implement `examples/canonical_plugin.lua`:**

Per CONTEXT.md specifics: "The canonical plugin example should demonstrate the full lifecycle: init with ctx, component registration, system registration, event handling, service usage, and the shutdown no-op stub."

```lua
--- Canonical Plugin Example
--- Reference implementation for all game plugins.
--- Demonstrates: init, component usage, system query, event handling,
--- service registration, and shutdown stub.

local evolved = require("lib.evolved")
local Components = require("src.core.components")

local CanonicalPlugin = {}

-- Plugin metadata
CanonicalPlugin.name = "canonical"
CanonicalPlugin.deps = {}  -- no dependencies for the example

function CanonicalPlugin:init(ctx)
    self.bus = ctx.bus
    self.worlds = ctx.worlds

    -- 1. Component usage — use shared components from src/core/components.lua
    -- (Components are defined centrally, not per-plugin)

    -- 2. System registration — create a query for Position+Velocity entities
    self._movement_query = evolved.builder()
        :include(Components.Position, Components.Velocity)
        :include(ctx.worlds.server.tag)
        :build()

    -- 3. Event handling — subscribe to events
    ctx.bus:on("entity_spawned", function(data)
        -- Example handler: log or react to entity spawn
        self._last_spawned = data.entity
    end)

    -- 4. Service registration (example: expose a query service)
    ctx.services:register("canonical_query", {
        get_movement_query = function()
            return self._movement_query
        end,
    })
end

--- System update — called each tick by the game loop
function CanonicalPlugin:update(dt)
    -- Process movement query
    for chunk, _entities, count in evolved.execute(self._movement_query) do
        local positions = chunk:components(Components.Position)
        local velocities = chunk:components(Components.Velocity)
        for i = 1, count do
            positions[i].x = positions[i].x + velocities[i].dx * dt
            positions[i].y = positions[i].y + velocities[i].dy * dt
        end
    end
end

--- Shutdown stub — contract established for future use
function CanonicalPlugin:shutdown(_ctx)
    -- No-op: cleanup will be implemented when save/load or resource management requires it
end

return CanonicalPlugin
```

**Update `main.lua`** to delegate to the plugin registry boot sequence:

```lua
local Registry = require("src.core.registry")
local Bus = require("src.core.bus")
local Worlds = require("src.core.worlds")
local Context = require("src.core.context")
local plugin_list = require("src.core.plugin_list")

local _registry
local _ctx

function love.load()
    local bus = Bus.new()
    local worlds = Worlds.create()
    _ctx = Context.new({ worlds = worlds, bus = bus })

    _registry = Registry.new()

    -- Register plugins from the explicit plugin list
    for _, entry in ipairs(plugin_list) do
        local plugin_module = require(entry.module)
        _registry:register(entry.name, plugin_module, { deps = entry.deps or {} })
    end

    -- Boot all plugins (topological order, fail-fast on missing deps)
    _registry:boot(_ctx)
end

function love.update(dt)
    -- Future: call registered systems update
    -- Flush event bus at end of tick
    if _ctx then
        _ctx.bus:flush()
    end
end

function love.draw()
    -- Future: call registered render systems
end
```

No game logic in love callbacks — only infrastructure delegation. The plugin_list is empty so boot is currently a no-op, but the wiring is complete.
