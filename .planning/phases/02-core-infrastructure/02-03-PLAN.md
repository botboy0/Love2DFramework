```lua
local binser = require("lib.binser")

local Transport = {}
Transport.__index = Transport

local DEFAULT_WARNING_THRESHOLD = 100  -- per CONTEXT.md: Claude's discretion on exact value

function Transport.new(opts)
    opts = opts or {}
    return setmetatable({
        _outbound = opts.outbound_channel,  -- injectable: love.thread.Channel or mock
        _inbound = opts.inbound_channel,    -- injectable: love.thread.Channel or mock
        _queue = {},                        -- outbound queue
        _networkable = {},                  -- set of event names that cross the boundary
        _warning_threshold = opts.warning_threshold or DEFAULT_WARNING_THRESHOLD,
    }, Transport)
end

function Transport:mark_networkable(event_name) ... end
function Transport:is_networkable(event_name) ... end
function Transport:queue(event_name, data) ... end  -- only queues if networkable
function Transport:flush() ... end                   -- serialize + push all queued
function Transport:receive() ... end                 -- pop + deserialize one message
function Transport:receive_all() ... end             -- pop all available messages
```

**Channel interface** (duck-typed, works with love.thread.Channel or mock):
- `channel:push(value)` — enqueue a value
- `channel:pop()` — dequeue a value (returns nil if empty)
- `channel:getCount()` — number of items in channel

**Mock channel for tests:**
```lua
local function mock_channel()
    local items = {}
    return {
        push = function(self, v) table.insert(items, v) end,
        pop = function(self) return table.remove(items, 1) end,
        getCount = function(self) return #items end,
    }
end
```

**Message format:** Each message is a table `{event = "event_name", data = {...}}` serialized to a string via `binser.serialize()`.

**Per CONTEXT.md decisions:**
- Flush once per tick, aligned with event bus flush
- Explicit opt-in for networkable events
- Warn on queue backup (log with print), don't block or drop
- Channel naming conventions: Claude's discretion — use `"server_to_client"` and `"client_to_server"`
